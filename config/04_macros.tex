% 1b. Rutas, endpoints y rutas de archivo con quiebre automático de línea.
% Evita el desbordamiento (overfull \hbox) de cadenas largas sin espacios,
% como /api/v1/work-experiences o @/shared/services/apiClient.ts. Se permite
% el quiebre tras cada separador (/  _  -  .  @) mediante \discretionary.
\newcommand{\ruta}[1]{{\ttfamily\fontsize{8.5}{10.2}\selectfont\rutaBreak{#1}}}
\newcommand{\rutaBreak}[1]{\seqsplit{#1}}

% 1c. Versión en línea más legible: respeta los caracteres pero inserta puntos
% de quiebre lógicos tras los separadores habituales de rutas/identificadores.
\makeatletter
\newcommand{\pathbreak}[1]{\texttt{\@pathbreak#1\relax}}
\def\@pathbreak#1{%
  \ifx#1\relax\else
    \ifx#1/\,\discretionary{}{}{}/\discretionary{}{}{}%
    \else\ifx#1_\_\discretionary{}{}{}%
    \else\ifx#1-#1\discretionary{}{}{}%
    \else\ifx#1.#1\discretionary{}{}{}%
    \else#1\fi\fi\fi\fi
    \expandafter\@pathbreak
  \fi}
\makeatother

% 1d. Tipos de columna con CENTRADO VERTICAL de contenido en cada celda.
% (m{} de array centra verticalmente; \arraybackslash restaura \\ tras
% un comando de alineación horizontal.)
\newcolumntype{L}[1]{>{\raggedright\arraybackslash}m{#1}}  % izq. + centro vert.
\newcolumntype{C}[1]{>{\centering\arraybackslash}m{#1}}    % centro + centro vert.
\newcolumntype{R}[1]{>{\raggedleft\arraybackslash}m{#1}}   % der. + centro vert.
% Variante de X (tabularx) con centrado vertical de la celda.
\newcolumntype{Y}{>{\raggedright\arraybackslash}X}
% \celdaCentro: centra vertical y horizontalmente un dato corto en columnas p/X.

% 4c. Diccionario de Datos (longtable multipágina por tabla)
% Uso: \begin{DiccTabla}{nombre_tabla} filas... \end{DiccTabla}
% Nota: las columnas usan \raggedright\arraybackslash para permitir el quiebre
% de línea dentro de la celda y evitar el desbordamiento (overfull) horizontal.
% Para cadenas largas sin espacios dentro de una celda (rutas, enums separados
% por '/' o '_') se usa \pathbreak/\seqsplit en la fila concreta.
\newenvironment{DiccTabla}[1]{%
    \renewcommand{\arraystretch}{1.3}%
    \fontsize{7.6}{9.2}\selectfont\sffamily%
    \setlength{\tabcolsep}{4pt}%
    \begin{longtable}{%
        |>{\ttfamily\bfseries\color{colorPrimario}\raggedright\arraybackslash}p{2.7cm}%
        |>{\ttfamily\raggedright\arraybackslash}p{2.5cm}%
        |c%
        |>{\ttfamily\scriptsize\raggedright\arraybackslash}p{2.3cm}%
        |>{\raggedright\arraybackslash}p{3.5cm}|}
    \caption{Diccionario de datos — \texttt{core.#1}}\\
    \hline
    \rowcolor{headerTabla}
    \sffamily\textbf{Columna} & \sffamily\textbf{Tipo} & \sffamily\textbf{Nulo} & \sffamily\textbf{Defecto} & \sffamily\textbf{Descripción} \\
    \hline
    \endfirsthead
    \multicolumn{5}{l}{\sffamily\footnotesize\itshape\color{grisISO}\texttt{core.#1} (continuación)}\\
    \hline
    \rowcolor{headerTabla}
    \sffamily\textbf{Columna} & \sffamily\textbf{Tipo} & \sffamily\textbf{Nulo} & \sffamily\textbf{Defecto} & \sffamily\textbf{Descripción} \\
    \hline
    \endhead
    \hline
    \multicolumn{5}{r}{\sffamily\tiny\color{grisISO}\textit{continúa en la página siguiente}}\\
    \endfoot
    \hline
    \endlastfoot
}{%
    \end{longtable}%
}
